```go
    SessionID      string `json:"session_id"`
    TranscriptPath string `json:"transcript_path"`

type PreToolUseInput struct {
    HookInputBase
    PermissionMode string          `json:"permission_mode"`
    ToolName       string          `json:"tool_name"`
    ToolInput      json.RawMessage `json:"tool_input"`
    ToolUseID      string          `json:"tool_use_id"`

    HookInputBase
    PermissionMode string          `json:"permission_mode"`

    ToolInput      json.RawMessage `json:"tool_input"`
    ToolResponse   json.RawMessage `json:"tool_response"`
    ToolUseID      string          `json:"tool_use_id"`
}

type HookOutput struct {
    Continue           *bool  `json:"continue,omitempty"`
    StopReason         string `json:"stopReason,omitempty"`
    SuppressOutput     bool   `json:"suppressOutput,omitempty"`
    AdditionalContext  string `json:"additionalContext,omitempty"`
    HookSpecificOutput any    `json:"hookSpecificOutput,omitempty"`
}

type PreToolUseHookOutput struct {
    HookEventName            string          `json:"hookEventName"`

    PermissionDecisionReason string          `json:"permissionDecisionReason,omitempty"`

    AdditionalContext        string          `json:"additionalContext,omitempty"`
}

type PostToolUseHookOutput struct {
    HookEventName     string `json:"hookEventName"`
    AdditionalContext string `json:"additionalContext,omitempty"`
}
```

    **Create internal/hook/pre_compact.go (per D-07, D-08, D-09, research Pattern 5).**

    ```go
    func handlePreCompact(ctx context.Context, raw json.RawMessage, hs *hookState, w io.Writer) error
    ```

    The function:
    1. Decodes raw into PreCompactInput.
    2. Builds a snapshot struct:
       ```go
       type compactSnapshot struct {
           SessionID      string `json:"session_id"`
           TranscriptPath string `json:"transcript_path"`
           Trigger        string `json:"trigger"`
           Timestamp      string `json:"timestamp"` // time.Now().UTC().Format(time.RFC3339)
           CWD            string `json:"cwd"`
       }
       ```
    3. Creates `.gsdw/` directory at `filepath.Join(input.CWD, ".gsdw")` with `os.MkdirAll(dir, 0755)`.
    4. Writes snapshot atomically using temp+rename pattern (same as graph/index.go):
       - `bufferPath := filepath.Join(gsdwDir, "precompact-snapshot.json")`
       - `tmp := bufferPath + ".tmp"`
       - `os.WriteFile(tmp, data, 0644)`
       - `os.Rename(tmp, bufferPath)`
    5. On any write error, log to `slog.Error("precompact: local write failed", "err", err)` but do NOT return error. PreCompact must never block compaction (D-07).
    6. Return `writeOutput(w, HookOutput{})` -- empty output, no context injection.

    **CRITICAL: No goroutines.** Per research Pitfall 2, the async Dolt sync pattern using goroutines is unsafe (process exits, goroutine killed). The local buffer IS the save. A future hook invocation (SessionStart or PostToolUse) can detect and sync the pending snapshot to Dolt.

    **Step 1: Create pre_tool_use.go (per D-10, D-11, research Pattern 6).**

    ```go
    // contextInjectTools are tools that modify state and benefit from bead context injection.
    var contextInjectTools = map[string]bool{
        "Write": true,
        "Edit":  true,
        "Bash":  true,
        "Agent": true,
    }

    func handlePreToolUse(ctx context.Context, raw json.RawMessage, hs *hookState, w io.Writer) error
    ```

    The function:
    1. Decodes raw into PreToolUseInput.
    2. Fast path: if `!contextInjectTools[input.ToolName]`, return `writeOutput(w, preToolUseAllow(""))` immediately. No graph queries for read-class tools.
    3. For write-class tools: attempt to load context from local index first (fast path <1ms):
       - `idx, err := graph.LoadIndex(filepath.Join(input.CWD, ".gsdw"))` -- if err, fall back to empty context.
       - If index loaded, build brief context string: "Active phase: {phase key}" from index.PhaseToID.
    4. If .beads/ exists and index loaded, try graph query with strict timeout (400ms per D-11):
       - `hs.beadsDir = input.CWD; hs.init(ctx)` -- on error, use index-only context.
       - `queryCtx, cancel := context.WithTimeout(ctx, 400*time.Millisecond)`
       - `ready, _ := hs.client.ListReady(queryCtx)` -- on error, use partial context.
       - Append ready task titles to context string.
    5. Return `writeOutput(w, preToolUseAllow(context))`.

    Helper:
    ```go
    func preToolUseAllow(additionalContext string) HookOutput {
        return HookOutput{
            HookSpecificOutput: PreToolUseHookOutput{
                HookEventName:      "PreToolUse",
                PermissionDecision: "allow",
                AdditionalContext:  additionalContext,
            },
        }
    }
    ```
