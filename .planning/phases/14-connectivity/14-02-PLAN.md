          ```
          gsdwDir := opts.findGsdwDirFn()

          ```

          **Phase 2: Check existing config (D-05)**
          ```
          existing, _ := opts.loadConfigFn(gsdwDir)
          if existing != nil {
              host, port := existing.ActiveHostPort()
              print "Current connection: {mode} ({host}:{port})"
              healthErr := opts.healthCheckFn(host, port, existing.Remote.User, os.Getenv("GSDW_DB_PASSWORD"))
              if healthErr != nil { print "[FAIL] {healthErr}" } else { print "[OK]   Server responding" }
              prompt "Reconfigure? [y/N]: "
              if answer != "y" { return nil }
          }
          ```

          **Phase 3: Auto-detect local server (D-01)**
          ```
          print "Scanning for running Dolt server on 127.0.0.1:3307..."
          if opts.detectServerFn("127.0.0.1", "3307") == nil {
              print "Found local Dolt server on 127.0.0.1:3307"
              prompt "Use it? [Y/n]: "
              if answer == "" || answer == "y" {
                  return saveLocalConfig(opts, gsdwDir, "127.0.0.1", "3307")
              }
          }
          ```

          **Phase 4: No server found — offer choices (D-02)**
          ```
          print "No running Dolt server found.\n"
          print "  1) Start local container\n"
          print "  2) Configure remote host\n"
          print "  3) Cancel\n"
          prompt "Choose [1-3]: "

              "2": call handleConfigureRemote(opts, gsdwDir, reader)
              "3": return nil
          ```

          - On failure (D-09, D-10, D-11): call `handleRemoteFallback(opts, gsdwDir, reader, host, port, user, healthErr)`

       g. `handleRemoteFallback(opts, gsdwDir, reader, host, port, user, healthErr)` (D-09/D-10/D-11):
          - Print "[FAIL] {healthErr}"
          - Prompt "Fall back to local container? [y/N]: " (D-09 — blocks, no timeout)
          - If "n" or empty: return error wrapping healthErr
          - If "y":
            - Call `opts.startContainerFn()` (D-10 — start if not running)
            - On start failure: return error
            - Prompt "Make this the default? [y/N]: " (D-11)
            - If "y": save config with active_mode="local", preserve remote settings in Remote field
            - If "n": save config with active_mode="remote" (preserved), but session already uses local (print note)

       h. `saveLocalConfig(opts, gsdwDir, host, port)`:
          - Build Config{ActiveMode: "local", Local: {Host: host, Port: port}, Configured: time.Now().UTC().Format(time.RFC3339)}
          - Call opts.saveConfigFn
          - Print "Connection saved to .gsdw/connection.json"
